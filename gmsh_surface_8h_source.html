26 public:

30  static void reset()
31  {
32  auto it = allGmshSurfaces.begin();
33  for(; it != allGmshSurfaces.end(); ++it) {
34  if(!it->second->vertex_defined_on_surface) delete it->second;

36  allGmshSurfaces.clear();
37  };
38  static int maxTag()
39  {
40  int t = 0;

43  }
