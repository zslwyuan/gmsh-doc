32  // compose an integration rule for (1-w)^2 f(u,v,w) on the standard
33  // hexahedron
34 
35  int iW = i / (nbPtUV2);
36  int iU = (i - iW * nbPtUV2) / nbPtUV;

44

48  intpt[l].pt[1] = 0.5 * (1 - wp) * vp;
49  intpt[l].pt[2] = 0.5 * (1 + wp);
