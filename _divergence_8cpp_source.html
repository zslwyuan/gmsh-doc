60  if(numComp != 3) continue;

63  std::vector<double> *out = data2->incrementList(1, type, numNodes);
64  if(!out) continue;

66  for(int nod = 0; nod < numNodes; nod++)

68  z[nod]);

72  if(!element) continue;

97  }
