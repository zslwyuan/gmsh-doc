99 IntPt *getGQQPts(int order, bool forceTensorRule)
100 {
101  if(!forceTensorRule && order <= 2) return GQQ[order];
102 
103  if(static_cast<int>(GQQGL.size()) < order + 1)
104  GQQGL.resize(order + 1, nullptr);
105  if(!GQQGL[order]) {
106  int n = (order + 1) / (float)2 + 0.5;
107  double *pt, *wt;
108  gmshGaussLegendre1D(n, &pt, &wt);
109  IntPt *intpt = new IntPt[n * n];
110  int k = 0;
111  for(int i = 0; i < n; i++) {
112  for(int j = 0; j < n; j++) {
113  intpt[k].pt[0] = pt[i];
114  intpt[k].pt[1] = pt[j];
115  intpt[k].pt[2] = 0.0;
116  intpt[k++].weight = wt[i] * wt[j];
117  }
118  }
119  GQQGL[order] = intpt;
120  }
121  return GQQGL[order];
122 }

124 int getNGQQPts(int order, bool forceTensorRule)
