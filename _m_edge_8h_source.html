44  int computeCorrespondence(MEdge &other)
45  {
46  if(other.getVertex(0) == _v[0] && other.getVertex(1) == _v[1])
47  return 1;
48  else if(other.getVertex(0) == _v[1] && other.getVertex(1) == _v[0])
49  return -1;
50  return 0;
51  }

83  {

86  if(t[0] == 0.)
