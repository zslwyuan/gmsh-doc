126  // compute the normal of an internal vertex v
127  inline SVector3 normal(const Vertex *v) const
128  {
129  SVector3 n(0, 0, 0);
130  HalfEdge *he = v->he;
131  do {
132  SVector3 n1 = he->d();
133  he = he->opposite;
134  if(he == NULL) return -1;
135  he = he->next;
136  n += crossprod(n1, he->d());
