19 // Hash FNV1a implemented via for loop. "key" has size "len" bytes.

21 inline size_t hash_FNV1a(const void *const key, const int len)
22 {
23  const unsigned char *p = static_cast<const unsigned char *>(key);
24  size_t hash = FNV_OFFSET_BASIS;
25  for(int n = len; n--;) hash = (hash ^ static_cast<size_t>(*p++)) * FNV_PRIME;
26  return hash;
27 }
