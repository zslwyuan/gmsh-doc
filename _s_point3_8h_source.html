78  bool transform(const std::vector<double> &tfo)
79  {
80  if(tfo.size() < 12) return false;
81  double old[3] = {P[0], P[1], P[2]};
82  P[0] = P[1] = P[2] = 0.;
83  int idx = 0;
84  for(int i = 0; i < 3; i++) {
85  for(int j = 0; j < 3; j++) P[i] += old[j] * tfo[idx++];
86  P[i] += tfo[idx++];
87  }
88  return true;
89  }
